1// SWIM protocol implementation.
2//
3// Reference: Das et al., "SWIM: Scalable Weakly-consistent Infection-style
4// Process Group Membership Protocol", DSN 2002.
5//
6// Architecture:
7// Two threads — a sender (periodic PING loop) and a receiver (UDP listen loop).
8// They communicate via a pending_acks map with condition_variable notification.
9// The sender waits for ACKs; the receiver deposits them.
10//
11// Failure detection pipeline (per protocol round):
12// 1. Select random alive peer → send PING
13// 2. If ACK received within T_ping → peer alive, done
14// 3. If no ACK → select k random proxies → send PING_REQ (indirect probe)
15// 4. If any proxy relays an ACK → peer alive, done
16// 5. If no indirect ACK → mark peer SUSPECT
17// 6. If peer stays SUSPECT for T_suspect → mark DEAD
18// 7. If peer receives its own SUSPECT → increment incarnation → refute
19//
20// Dissemination:
21// Membership updates are piggybacked on every outgoing message (PING, ACK,
22// PING_REQ). This provides O(log N) convergence without dedicated gossip
23// rounds, using the protocol's existing message traffic as a carrier.

73// Graceful leave: broadcast a DEAD message about ourselves so peers can
74// remove us instantly without waiting for the suspect timeout. This is an
75// optimization for rolling updates — drain → graceful leave → restart is
76// significantly faster than drain → crash detection → restart.
77void SwimProtocol::LeaveCluster() {
78 auto peers = membership_list_.GetAliveMembers();
79
80 // Use incarnation + 1 to ensure this overrides any concurrent ALIVE
81 // messages that might be in-flight from our own piggybacked updates.
82 MembershipUpdate leave_update;
83 leave_update.set_member_id(my_id_);
84 leave_update.set_address(my_address_);
85 leave_update.set_state(DEAD);
86 leave_update.set_incarnation(membership_list_.my_incarnation() + 1);
87
88 for (const auto& [peer_id, peer_info] : peers) {
89 GossipMessage msg;
90 msg.set_type(PING);
91 msg.set_sender_id(my_id_);
92 msg.set_sequence_num(next_seq_num_.fetch_add(1));
93 *msg.add_updates() = leave_update;
94
95 Address dest = ParseAddress(peer_info.address);
96 transport_.Send(dest, msg);
97 }
98
99 LOG_INFO("swim", "%s sent graceful leave to %zu peers", my_id_.c_str(), peers.size());
100}

127// ============================================================================
128// Sender thread
129//
130// Runs one "protocol round" every T_protocol milliseconds. Each round:
131// 1. Pings a random peer (direct probe)
132// 2. If no ACK, tries indirect probes through k other peers
133// 3. If still no ACK, marks the peer as SUSPECT
134// 4. Scans suspect list and promotes timed-out suspects to DEAD
135// ============================================================================
136
137void SwimProtocol::SenderLoop() {
138 while (running_.load()) {
139 auto round_start = std::chrono::steady_clock::now();
140
141 DoPingRound();
142 CheckSuspectTimeouts();
143
144 // Maintain consistent round timing by sleeping for the remainder.
145 // If the round took longer than T_protocol (e.g., due to slow indirect
146 // probes), the next round starts immediately with no sleep.
147 auto elapsed = std::chrono::steady_clock::now() - round_start;
148 auto sleep_time = std::chrono::milliseconds(config_.protocol_period_ms) - elapsed;
149 if (sleep_time > std::chrono::milliseconds(0)) {
150 std::this_thread::sleep_for(sleep_time);
151 }
152 }
153}

155void SwimProtocol::DoPingRound() {
156 // Step 1: Select a random alive peer to probe.
157 auto peer = membership_list_.GetRandomAlivePeer();
158 if (!peer) return;
159
160 const auto& [peer_id, peer_info] = *peer;
161 uint64_t seq = next_seq_num_.fetch_add(1);
162
163 // Register the pending ACK before sending, so the receiver thread can
164 // record it even if the response arrives very quickly.
165 {
166 std::lock_guard lock(ack_mutex_);
167 pending_acks_[seq] = false;
168 }
169
170 // Step 2: Send direct PING.
171 Address dest = ParseAddress(peer_info.address);
172 SendMessage(dest, PING, "", seq);
173
174 // Step 3: Wait for direct ACK.
175 if (WaitForAck(seq, config_.ping_timeout_ms)) {
176 std::lock_guard lock(ack_mutex_);
177 pending_acks_.erase(seq);
178 return; // Peer is alive.
179 }
180
181 // Step 4: Direct ping failed — try indirect probes.
182 // Ask k random other alive peers to ping the suspect on our behalf.
183 // This handles cases where the direct network path is temporarily broken
184 // but the suspect is reachable through other nodes.
185 std::vector<std::string> exclude = {peer_id};
186 std::vector<uint64_t> indirect_seqs;
187
188 for (int i = 0; i < config_.indirect_ping_count; i++) {
189 auto proxy = membership_list_.GetRandomAlivePeer(exclude);
190 if (!proxy) break;
191
192 uint64_t indirect_seq = next_seq_num_.fetch_add(1);
193 {
194 std::lock_guard lock(ack_mutex_);
195 pending_acks_[indirect_seq] = false;
196 }
197
198 Address proxy_addr = ParseAddress(proxy->second.address);
199 SendMessage(proxy_addr, PING_REQ, peer_id, indirect_seq);
200 indirect_seqs.push_back(indirect_seq);
201
202 exclude.push_back(proxy->first);
203 }
204
205 // Step 5: Wait for any indirect ACK (any one is enough to confirm alive).
206 bool got_indirect_ack = false;
207 if (!indirect_seqs.empty()) {
208 auto deadline = std::chrono::steady_clock::now() +
209 std::chrono::milliseconds(config_.ping_timeout_ms);
210
211 std::unique_lock lock(ack_mutex_);
212 ack_cv_.wait_until(lock, deadline, [&]() {
213 for (uint64_t s : indirect_seqs) {
214 if (pending_acks_.count(s) && pending_acks_[s]) return true;
215 }
216 return false;
217 });
218
219 for (uint64_t s : indirect_seqs) {
220 if (pending_acks_.count(s) && pending_acks_[s]) {
221 got_indirect_ack = true;
222 }
223 pending_acks_.erase(s);
224 }
225 }
226
227 {
228 std::lock_guard lock(ack_mutex_);
229 pending_acks_.erase(seq);
230 }
231
232 if (got_indirect_ack) {
233 return; // Indirect probe succeeded — peer is alive.
234 }
235
236 // Step 6: Both direct and indirect probes failed — mark SUSPECT.
237 // The peer gets T_suspect to refute (via incarnation increment) before
238 // being declared DEAD.
239 membership_list_.MarkSuspect(peer_id);
240}

242// Scan the suspect list and promote any member that has been SUSPECT for
243// longer than T_suspect to DEAD. This runs once per protocol round.
244void SwimProtocol::CheckSuspectTimeouts() {
245 auto suspects = membership_list_.GetMembersByState(SUSPECT);
246 auto now = std::chrono::steady_clock::now();
247
248 for (const auto& [id, info] : suspects) {
249 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
250 now - info.suspect_time);
251 if (elapsed.count() >= config_.suspect_timeout_ms) {
252 membership_list_.MarkDead(id);
253 }
254 }
255}

258// Receiver thread
259//
260// Listens for incoming UDP messages and dispatches to the appropriate handler.

264void SwimProtocol::ReceiverLoop() {
265 while (running_.load()) {
266 auto result = transport_.Receive(100);
267 if (!result) continue;

269 auto& [msg, sender_addr] = *result;
270 HandleMessage(msg, sender_addr);
271 }

274void SwimProtocol::HandleMessage(const GossipMessage& msg,
275 const Address& sender_addr) {
276 // Always process piggybacked membership updates first, regardless of
277 // message type. This is how SWIM achieves epidemic dissemination —
278 // every message carries state updates as a "free" side channel.
279 ProcessPiggybackedUpdates(msg);
280
281 switch (msg.type()) {
282 case PING:
283 HandlePing(msg, sender_addr);
284 break;
285 case PING_REQ:
286 HandlePingReq(msg, sender_addr);
287 break;
288 case ACK:
289 HandleAck(msg);
290 break;
291 default:
292 break;
293 }
294}

296// PING handler: learn the sender's identity and reply with ACK.
297// The sender's real ID is extracted from the message, replacing any
298// temporary "seed-<addr>" ID we may have assigned during bootstrap.
299void SwimProtocol::HandlePing(const GossipMessage& msg,
300 const Address& sender_addr) {
301 if (!msg.sender_id().empty()) {
302 membership_list_.AddMember(msg.sender_id(), sender_addr.ToString());
303 }
304 SendMessage(sender_addr, ACK, "", msg.sequence_num());
305}
306
307// PING_REQ handler: we are acting as a proxy for indirect probing.
308// Forward a PING to the target, and if we get an ACK, relay it back
309// to the original requester. This enables failure detection even when
310// the direct path between the requester and target is broken.
311void SwimProtocol::HandlePingReq(const GossipMessage& msg,
312 const Address& sender_addr) {
313 const auto& target_id = msg.target_id();
314 auto target_info = membership_list_.GetMember(target_id);
315 if (!target_info) return;
316
317 uint64_t probe_seq = next_seq_num_.fetch_add(1);
318 {
319 std::lock_guard lock(ack_mutex_);
320 pending_acks_[probe_seq] = false;
321 }
322
323 Address target_addr = ParseAddress(target_info->address);
324 SendMessage(target_addr, PING, "", probe_seq);
325
326 if (WaitForAck(probe_seq, config_.ping_timeout_ms)) {
327 // Target is alive — relay ACK back to the original requester
328 // using their original sequence number so they can match it.
329 SendMessage(sender_addr, ACK, "", msg.sequence_num());
330 }
331
332 std::lock_guard lock(ack_mutex_);
333 pending_acks_.erase(probe_seq);
334}

340// Process piggybacked membership updates from an incoming message.
341// Before applying each update, check if it's a suspicion about ourselves
342// that needs to be refuted.
343void SwimProtocol::ProcessPiggybackedUpdates(const GossipMessage& msg) {
344 for (const auto& update : msg.updates()) {
345 CheckSelfRefutation(update);
346 membership_list_.ApplyUpdate(update);
347 }
348}
349
350// Self-refutation: if another node thinks we are SUSPECT or DEAD,
351// increment our incarnation number to override the false suspicion.
352// The incremented incarnation will be piggybacked on our next outgoing
353// message, propagating the refutation through the cluster.
354//
355// This is the key mechanism that prevents false positives in SWIM:
356// a slow-but-alive node can always clear its own suspicion by proving
357// it's still running (via a higher incarnation number).
358void SwimProtocol::CheckSelfRefutation(const MembershipUpdate& update) {
359 if (update.member_id() != my_id_) return;
360
361 if (update.state() == SUSPECT || update.state() == DEAD) {
362 if (update.incarnation() >= membership_list_.my_incarnation()) {
363 membership_list_.IncrementMyIncarnation();
364 LOG_INFO("swim", "%s refuting %s at incarnation %lu → %lu",
365 my_id_.c_str(),
366 update.state() == SUSPECT ? "SUSPECT" : "DEAD",
367 update.incarnation(),
368 membership_list_.my_incarnation());
369 }
370 }
371}

373// ============================================================================
374// Message sending
375//
376// Every outgoing message carries two types of piggybacked data:
377// 1. Recent membership updates from the piggyback buffer (epidemic dissemination)
378// 2. Our own load metadata (active_requests, max_capacity, model_version)
379//
380// This means every PING, ACK, and PING_REQ doubles as a gossip vector,
381// spreading membership and load information without dedicated gossip rounds.
382// ============================================================================
383
384void SwimProtocol::SendMessage(const Address& dest, MessageType type,
385 const std::string& target_id, uint64_t seq_num) {
386 GossipMessage msg;
387 msg.set_type(type);
388 msg.set_sender_id(my_id_);
389 msg.set_target_id(target_id);
390 msg.set_sequence_num(seq_num);
391
392 // Attach recent membership updates for epidemic dissemination.
393 auto updates = membership_list_.GetUpdatesForPiggyback(
394 config_.max_piggyback_updates);
395 for (auto& update : updates) {
396 *msg.add_updates() = std::move(update);
397 }
398
399 // Attach our own load metadata so the gateway can make routing decisions
400 // based on each replica's self-reported load.
401 {
402 std::lock_guard lock(load_mutex_);
403 auto* self_update = msg.add_updates();
404 self_update->set_member_id(my_id_);
405 self_update->set_address(my_address_);
406 self_update->set_state(ALIVE);
407 self_update->set_incarnation(membership_list_.my_incarnation());
408 self_update->set_active_requests(my_active_requests_);
409 self_update->set_max_capacity(my_max_capacity_);
410 self_update->set_model_version(my_model_version_);
411 }
412
413 transport_.Send(dest, msg);
414}

416// ============================================================================
417// ACK coordination between sender and receiver threads
418//
419// The sender registers a pending ACK (seq_num → false), sends a message,
420// then calls WaitForAck which blocks on a condition_variable.
421// The receiver calls RecordAck when an ACK arrives, setting the flag to
422// true and notifying the condition_variable.
423// ============================================================================
424
425bool SwimProtocol::WaitForAck(uint64_t seq_num, int timeout_ms) {
426 auto deadline = std::chrono::steady_clock::now() +
427 std::chrono::milliseconds(timeout_ms);
428
429 std::unique_lock lock(ack_mutex_);
430 return ack_cv_.wait_until(lock, deadline, [&]() {
431 return pending_acks_.count(seq_num) && pending_acks_[seq_num];
432 });
433}
434
435void SwimProtocol::RecordAck(uint64_t seq_num) {
436 std::lock_guard lock(ack_mutex_);
437 if (pending_acks_.count(seq_num)) {
438 pending_acks_[seq_num] = true;
439 ack_cv_.notify_all();
440 }
441}
