18// ---------------------------------------------------------------------------
19// ReplicaInfo: metadata for a single LLM inference replica.
20//
21// Each replica is identified by a unique replica_id (e.g., "replica-1").
22// The struct tracks two network addresses:
23// - gossip_address: the UDP endpoint used by the SWIM protocol for
24// failure detection (heartbeats, pings, acks).
25// - grpc_address: the TCP endpoint where the replica serves the
26// LLMReplica::Generate gRPC streaming API.
27//
28// Address derivation convention: the gateway derives grpc_address from
29// gossip_address automatically using the rule
30// gRPC port = gossip port - 10000
31// (e.g., gossip 17001 -> gRPC 7001). This avoids a separate discovery
32// channel; the gossip layer is the single source of membership truth
33// and gRPC addresses fall out deterministically from it.
34//
35// active_requests is an atomic counter updated in real-time by the gateway
36// on dispatch/completion, while reported_active_requests is a stale value
37// piggybacked on gossip messages from the replica itself. The load balancer
38// uses the atomic counter for local routing decisions.
39//
40// max_capacity bounds how many concurrent requests the replica will accept.
41// A value of 0 means "unlimited" (the replica self-manages its concurrency).
42// ---------------------------------------------------------------------------
43struct ReplicaInfo {
44 std::string grpc_address; // host:port for gRPC inference
45 std::string gossip_address; // host:port for gossip UDP
46 llmgateway::gossip::MemberState gossip_state = llmgateway::gossip::ALIVE;
47 double weight = 1.0; // capacity weight for load balancing
48 std::atomic<int> active_requests{0};
49 int32_t reported_active_requests = 0; // from gossip piggyback
50 int32_t max_capacity = 4;
51 std::string model_version;
52 bool draining = false; // set during rolling update
53
54 // Copy constructor is required because std::atomic is not copyable.
55 // Without this, containers (e.g., std::vector<ReplicaInfo>) would fail
56 // to compile. We explicitly .load() the atomic for a safe snapshot.
57 ReplicaInfo() = default;
58 ReplicaInfo(const ReplicaInfo& other)
59 : grpc_address(other.grpc_address),
60 gossip_address(other.gossip_address),
61 gossip_state(other.gossip_state),
62 weight(other.weight),
63 active_requests(other.active_requests.load()),
64 reported_active_requests(other.reported_active_requests),
65 max_capacity(other.max_capacity),
66 model_version(other.model_version),
67 draining(other.draining) {}
68};

70// ---------------------------------------------------------------------------
71// ReplicaRegistry: thread-safe, centralized view of all known replicas.
72//
73// This is the single source of truth for replica metadata within the gateway
74// process. It is populated/updated by the SWIM gossip layer (via callbacks)
75// and queried by the LoadBalancer on every routing decision.
76//
77// Thread-safety model:
78// - A std::shared_mutex protects the internal map.
79// - Read-heavy paths (GetAliveReplicas, GetReplica, Increment/Decrement)
80// acquire a shared (reader) lock so they never block each other.
81// - Write paths (UpdateFromGossip, SetDraining, SetGrpcAddress) acquire
82// an exclusive (writer) lock.
83// - The active_requests counter inside ReplicaInfo is std::atomic, so
84// IncrementActive/DecrementActive only need a shared lock (they mutate
85// the atomic, not the map itself).
86// ---------------------------------------------------------------------------
87class ReplicaRegistry {
88public:
89 // Update replica state from a gossip membership update.
90 // Called by the SWIM protocol callback whenever a membership event fires
91 // (join, suspect, dead, etc.). Inserts a new entry if the replica_id is
92 // not yet known, and derives the gRPC address from the gossip address
93 // if one has not been set explicitly.
94 void UpdateFromGossip(const std::string& replica_id,
95 const llmgateway::gossip::MembershipUpdate& update);
96
97 // Return a snapshot of replicas eligible for routing.
98 // Includes ALIVE and SUSPECT replicas (the load balancer may deprioritize
99 // SUSPECT ones). Excludes DEAD replicas, draining replicas, entries
100 // without a gRPC address, and non-replica nodes (seeds, gateways).
101 std::vector<std::pair<std::string, ReplicaInfo>> GetAliveReplicas() const;
102
103 // Get info for a specific replica. Returns nullopt if not found.
104 std::optional<ReplicaInfo> GetReplica(const std::string& replica_id) const;

106 // Increment/decrement the gateway-local active request counter.
