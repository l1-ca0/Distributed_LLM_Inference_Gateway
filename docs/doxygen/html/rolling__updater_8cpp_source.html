1// Rolling update orchestrator.

3// Coordinates the drain → stop → restart → rejoin sequence for each replica
4// in a rolling update. The key invariant is that at most one replica is
5// offline at any time, so the cluster always has N-1 replicas serving traffic.

19bool RollingUpdater::Update(const std::vector<std::string>& replica_ids,
20 const std::string& new_version,
21 StopCallback stop_fn,
22 RestartCallback restart_fn,
23 int drain_timeout_ms,
24 int rejoin_timeout_ms) {
25
26 for (const auto& replica_id : replica_ids) {
27 LOG_INFO("rolling", "updating %s to %s", replica_id.c_str(), new_version.c_str());
28
29 // Step 1: Drain — stop new requests, wait for in-flight to complete.
30 if (!DrainReplica(replica_id, drain_timeout_ms)) {
31 LOG_ERROR("rolling", "failed to drain %s", replica_id.c_str());
32 // Un-drain so routing resumes to this replica.
33 registry_.SetDraining(replica_id, false);
34 lb_.RebuildRing();
35 return false;
36 }
37
38 // Step 2: Stop — kill the replica process / stop gRPC + gossip.
39 LOG_INFO("rolling", "stopping %s", replica_id.c_str());
40 stop_fn(replica_id);
41
42 // Step 3: Restart — start a new instance with the new version.
43 LOG_INFO("rolling", "restarting %s with version %s",
44 replica_id.c_str(), new_version.c_str());
45 restart_fn(replica_id, new_version);
46
47 // Clear the draining flag and rebuild the ring so the restarted
48 // replica starts receiving traffic as soon as gossip marks it ALIVE.
49 // draining is a gateway-side flag that UpdateFromGossip does not
50 // touch, so we must explicitly reset it here after restart.
51 registry_.SetDraining(replica_id, false);
52 lb_.RebuildRing();
53
54 // Step 4: Wait for the restarted replica to rejoin via gossip.
55 if (!WaitForRejoin(replica_id, rejoin_timeout_ms)) {
56 LOG_WARN("rolling", "%s did not rejoin within timeout, continuing",
57 replica_id.c_str());
58 // Continue anyway — the replica may rejoin later via gossip.
59 }
60
61 LOG_INFO("rolling", "%s updated successfully", replica_id.c_str());
62 }
63
64 LOG_INFO("rolling", "rolling update complete: %zu replicas updated to %s",
65 replica_ids.size(), new_version.c_str());
66 return true;
67}

69// ---------------------------------------------------------------------------
70// DrainReplica: mark the replica as draining and send a Drain RPC.
71//
72// Two-step drain:
73// 1. Mark draining in the gateway's registry so the load balancer stops
74// routing new requests to this replica.
75// 2. Send a Drain RPC to the replica itself so it rejects any new
76// Generate calls and waits for in-flight requests to complete.
77//
78// The Drain RPC is a blocking call — it returns only when all in-flight
79// requests on the replica have finished. This guarantees that after
80// DrainReplica returns, it's safe to stop the replica without dropping
81// any requests.
82//
83// If the Drain RPC fails (e.g., replica already crashed), we still
84// consider the drain successful — there are no in-flight requests if
85// the replica is dead.
86// ---------------------------------------------------------------------------
87bool RollingUpdater::DrainReplica(const std::string& replica_id,
88 int timeout_ms) {
89 // Step 1: Mark draining in the gateway registry.
90 registry_.SetDraining(replica_id, true);
91 lb_.RebuildRing();
92
93 // Step 2: Send Drain RPC to the replica.
94 auto info = registry_.GetReplica(replica_id);
95 if (!info || info->grpc_address.empty()) {
96 LOG_WARN("rolling", "no gRPC address for %s, skipping Drain RPC",
97 replica_id.c_str());
98 return true; // Already gone — drain is trivially complete.
99 }
100
101 auto channel = grpc::CreateChannel(info->grpc_address,
102 grpc::InsecureChannelCredentials());
103 auto stub = LLMReplica::NewStub(channel);
104
105 DrainRequest request;
106 DrainResponse response;
107 grpc::ClientContext context;
108
109 // Set a deadline for the Drain RPC.
110 context.set_deadline(std::chrono::system_clock::now() +
111 std::chrono::milliseconds(timeout_ms));
112
113 grpc::Status status = stub->Drain(&context, request, &response);
114
115 if (status.ok() && response.success()) {
116 LOG_INFO("rolling", "%s drained successfully", replica_id.c_str());
117 return true;
118 }
119
120 if (status.error_code() == grpc::StatusCode::DEADLINE_EXCEEDED) {
121 LOG_WARN("rolling", "drain timeout for %s", replica_id.c_str());
122 return false;
123 }
124
125 // Drain RPC failed (replica may have crashed). That's OK — no in-flight
126 // requests if the replica is dead.
127 LOG_WARN("rolling", "drain RPC failed for %s: %s (treating as drained)",
128 replica_id.c_str(), status.error_message().c_str());
129 return true;
130}

132// ---------------------------------------------------------------------------
133// WaitForRejoin: poll the registry until the replica appears as ALIVE.
134//
135// After a restart, the replica needs time to:
136// 1. Start its gRPC server
137// 2. Start its SWIM gossip protocol
138// 3. Be discovered by the gateway's gossip subscriber
139// 4. Have its state updated in the registry
140// 5. Be added back to the hash ring
141//
142// We poll every 200ms until the replica is ALIVE or the timeout expires.
143// ---------------------------------------------------------------------------
144bool RollingUpdater::WaitForRejoin(const std::string& replica_id,
145 int timeout_ms) {
146 auto deadline = std::chrono::steady_clock::now() +
147 std::chrono::milliseconds(timeout_ms);
148
149 while (std::chrono::steady_clock::now() < deadline) {
150 auto info = registry_.GetReplica(replica_id);
151 if (info && info->gossip_state == gossip::ALIVE && !info->draining) {
152 LOG_INFO("rolling", "%s rejoined cluster", replica_id.c_str());
153 return true;
154 }
155 std::this_thread::sleep_for(std::chrono::milliseconds(200));
156 }
157
158 return false;
159}
