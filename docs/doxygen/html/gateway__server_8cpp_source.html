36// ---------------------------------------------------------------------------
37// Infer: the main request handler. This is the hot path for every client
38// inference call and implements the mid-stream failover strategy.
39//
40// Retry loop design:
41// The loop runs up to max_retries times. Each iteration:
42// 1. Check if the client cancelled (e.g., timeout, user abort).
43// 2. Ask the LoadBalancer for a replica, passing an exclude set of
44// replicas already known to be unusable for this request.
45// 3. Check the circuit breaker. If the selected replica's circuit is
46// OPEN, add it to the exclude set so the next iteration picks a
47// different replica.
48// 4. Increment the active-request counter (for load-balancer awareness),
49// then stream tokens from the replica.
50// 5. On success (streamed >= 0): accumulate tokens_sent, record a
51// circuit-breaker success. If all tokens are delivered, return OK.
52// 6. On replica failure (streamed == -1): record a circuit-breaker
53// failure, add the replica to the exclude set, and retry.
54// tokens_sent is NOT incremented, so the next replica picks up
55// from where we left off via tokens_already_generated.
56//
57// The exclude set is the key to robust retries: consistent-hashing selection
58// is otherwise deterministic for a given prompt, so without exclusions the
59// retry loop would keep picking the same broken replica every iteration.
60//
61// Partial delivery: if some tokens were sent before all retries exhausted,
62// return OK rather than an error. The client already received partial
63// output on the stream, so an error status would be confusing. The client
64// can detect incompleteness by checking whether is_final was ever true.
65//
66// RESOURCE_EXHAUSTED: returned when no replica has capacity at all (before
67// any retry). This signals the client to back off or queue locally.
68// ---------------------------------------------------------------------------
69grpc::Status GatewayServer::Infer(grpc::ServerContext* context,
70 const InferRequest* request,
71 grpc::ServerWriter<InferResponse>* writer) {
72 // Dispatch to hedged path if requested and there are at least 2 replicas.
73 if (request->hedge()) {
74 auto alive = registry_.GetAliveReplicas();
75 if (alive.size() >= 2) {
76 return InferHedged(context, request, writer);
77 }
78 // Fall through to normal path if not enough replicas for hedging.
79 }
80
81 const int max_tokens = request->max_tokens();
82 const std::string& prompt = request->prompt();
83 int tokens_sent = 0;
84 int max_retries = 3;
85
86 // Replicas to skip on subsequent retries for this request: circuits
87 // that tripped OPEN or streams that just failed. Consistent-hashing
88 // selection is deterministic for a given prompt, so without this set
89 // the retry loop would keep picking the same broken replica.
90 std::unordered_set<std::string> excluded;
91
92 for (int attempt = 0; attempt < max_retries && tokens_sent < max_tokens; attempt++) {
93 if (context->IsCancelled()) {
94 return grpc::Status::CANCELLED;
95 }
96
97 auto selection = lb_.SelectReplica(prompt, excluded);
98 if (!selection) {
99 // No replica has capacity — wait in the backpressure queue.
100 // The calling thread blocks until NotifyCapacityAvailable() is
101 // called (when another request finishes) or the timeout expires.
102 if (!queue_.WaitForCapacity()) {
103 return grpc::Status(grpc::StatusCode::RESOURCE_EXHAUSTED,
104 "all replicas at capacity and queue full");
105 }
106 // Retry selection after being woken.
107 selection = lb_.SelectReplica(prompt, excluded);
108 if (!selection) {
109 // Still no capacity (e.g., replica died while we were queued).
110 continue;
111 }
112 }
113
114 // Circuit breaker check: if the replica is in OPEN state, skip it
115 // for the remainder of this request and let the LB pick another.
116 if (!cb_manager_.AllowRequest(selection->replica_id)) {
117 LOG_WARN("gateway", "circuit open for %s, excluding and retrying",
118 selection->replica_id.c_str());
119 excluded.insert(selection->replica_id);
120 continue;
121 }
122
123 // Track this in-flight request so the load balancer accounts for it.
124 registry_.IncrementActive(selection->replica_id);
125
126 int streamed = StreamFromReplica(
127 selection->replica_id, selection->grpc_address,
128 prompt, max_tokens, tokens_sent, writer, context);
129
130 registry_.DecrementActive(selection->replica_id);
131
132 // Notify the backpressure queue that a slot has freed up.
133 // This wakes the next FIFO-waiting request thread (if any).
134 queue_.NotifyCapacityAvailable();
135
136 if (streamed < 0) {
137 // streamed == -1 signals a replica-side gRPC error. Record the
138 // failure so the circuit breaker can track the error rate and
139 // potentially open the circuit for this replica. Also exclude it
140 // from this request's retry set so we pick a different replica.
141 cb_manager_.RecordFailure(selection->replica_id);
142 excluded.insert(selection->replica_id);
143 LOG_WARN("gateway", "replica %s failed after %d tokens, retrying",
144 selection->replica_id.c_str(), tokens_sent);
145 continue;
146 }
147
148 tokens_sent += streamed;
149 cb_manager_.RecordSuccess(selection->replica_id);
150
151 if (tokens_sent >= max_tokens) {
152 return grpc::Status::OK;
153 }
154
155 // Partial stream (streamed > 0 but < remaining): the replica ended
156 // early. Don't retry the same replica — move on.
157 excluded.insert(selection->replica_id);
158 }
159
160 if (tokens_sent >= max_tokens) {
161 return grpc::Status::OK;
162 }
163
164 if (tokens_sent > 0) {
165 // Partial delivery: some tokens were already written to the stream.
166 // Returning an error here would be misleading since the client
167 // already received data. Return OK and let the client detect
168 // incompleteness via the absence of is_final=true.
169 LOG_WARN("gateway", "partial delivery: %d/%d tokens", tokens_sent, max_tokens);
170 return grpc::Status::OK;
171 }
172
173 return grpc::Status(grpc::StatusCode::UNAVAILABLE,
174 "all replicas failed after retries");
175}

177// ---------------------------------------------------------------------------
178// InferHedged: speculative execution — race two replicas.
179//
180// Strategy (inspired by Google's "The Tail at Scale"):
181// 1. Select two different replicas via the load balancer.
182// 2. Open Generate streams to both simultaneously (in separate threads).
183// 3. Whichever replica produces its first token faster becomes the "winner."
184// 4. Stream all remaining tokens from the winner to the client.
185// 5. Cancel the loser's stream to free its capacity.
186//
187// This reduces tail latency at the cost of extra replica utilization.
188// The winner is determined by first-token latency (time to first Read()),
189// not by which thread starts first. Both threads read one token, then
190// signal via a shared promise. The main thread picks the winner and
191// continues streaming from it.
192// ---------------------------------------------------------------------------
193grpc::Status GatewayServer::InferHedged(grpc::ServerContext* context,
194 const InferRequest* request,
195 grpc::ServerWriter<InferResponse>* writer) {
196 const int max_tokens = request->max_tokens();
197 const std::string& prompt = request->prompt();
198
199 // Select two different replicas.
200 auto sel1 = lb_.SelectReplica(prompt);
201 if (!sel1) {
202 return grpc::Status(grpc::StatusCode::RESOURCE_EXHAUSTED,
203 "no replica available for hedging");
204 }
205
206 // For the second replica, we need a different one. Temporarily increment
207 // the first replica's active count so the LB deprioritizes it.
208 registry_.IncrementActive(sel1->replica_id);
209 auto sel2 = lb_.SelectReplica(prompt + "_hedge"); // different hash to get different replica
210 registry_.DecrementActive(sel1->replica_id);
211
212 if (!sel2 || sel2->replica_id == sel1->replica_id) {
213 // Only one replica available — fall through to non-hedged path.
214 // Re-select since we decremented.
215 registry_.IncrementActive(sel1->replica_id);
216 int streamed = StreamFromReplica(sel1->replica_id, sel1->grpc_address,
217 prompt, max_tokens, 0, writer, context);
218 registry_.DecrementActive(sel1->replica_id);
219 queue_.NotifyCapacityAvailable();
220 if (streamed >= 0) {
221 cb_manager_.RecordSuccess(sel1->replica_id);
222 return grpc::Status::OK;
223 }
224 cb_manager_.RecordFailure(sel1->replica_id);
225 return grpc::Status(grpc::StatusCode::UNAVAILABLE, "replica failed");
226 }
227
228 LOG_INFO("gateway", "hedging: racing %s vs %s",
229 sel1->replica_id.c_str(), sel2->replica_id.c_str());
230
231 registry_.IncrementActive(sel1->replica_id);
232 registry_.IncrementActive(sel2->replica_id);
233
234 // Shared state for the race: each thread reads one token, then signals.
235 struct RaceResult {
236 std::string replica_id;
237 std::string grpc_address;
238 GenerateResponse first_token;
239 std::unique_ptr<grpc::ClientContext> ctx;
240 std::unique_ptr<grpc::ClientReader<GenerateResponse>> reader;
241 std::unique_ptr<LLMReplica::Stub> stub;
242 bool success = false;
243 };
244
245 auto race1 = std::make_shared<RaceResult>();
246 auto race2 = std::make_shared<RaceResult>();
247 race1->replica_id = sel1->replica_id;
248 race1->grpc_address = sel1->grpc_address;
249 race2->replica_id = sel2->replica_id;
250 race2->grpc_address = sel2->grpc_address;
251
252 // Promise: first thread to read a token sets it.
253 auto winner_promise = std::make_shared<std::promise<int>>(); // 1 or 2
254 auto winner_future = winner_promise->get_future();
255 auto winner_set = std::make_shared<std::atomic<bool>>(false);
256
257 // Lambda to open a stream and read the first token.
258 auto race_fn = [&](std::shared_ptr<RaceResult> result, int id,
259 std::shared_ptr<std::promise<int>> promise,
260 std::shared_ptr<std::atomic<bool>> flag) {
261 auto channel = grpc::CreateChannel(result->grpc_address,
262 grpc::InsecureChannelCredentials());
263 result->stub = LLMReplica::NewStub(channel);
264 result->ctx = std::make_unique<grpc::ClientContext>();
265
266 GenerateRequest gen_req;
267 gen_req.set_request_id("hedge-" + std::to_string(id));
268 gen_req.set_prompt(prompt);
269 gen_req.set_max_tokens(max_tokens);
270 gen_req.set_tokens_already_generated(0);
271
272 result->reader = result->stub->Generate(result->ctx.get(), gen_req);
273
274 // Read first token.
275 if (result->reader->Read(&result->first_token)) {
276 result->success = true;
277 // Try to be the winner.
278 bool expected = false;
279 if (flag->compare_exchange_strong(expected, true)) {
280 promise->set_value(id);
281 }
282 }
283 };
284
285 // The race threads capture `prompt` and `max_tokens` by reference
286 // from this stack frame, so both must be joined before InferHedged
287 // returns. The RAII guards enforce this on every exit path including
288 // exceptions; the explicit join() calls below handle the normal path.
289 struct ThreadJoiner {
290 std::thread& t;
291 ~ThreadJoiner() { if (t.joinable()) t.join(); }
292 };
293
294 std::thread t1(race_fn, race1, 1, winner_promise, winner_set);
295 ThreadJoiner j1{t1};
296 std::thread t2(race_fn, race2, 2, winner_promise, winner_set);
297 ThreadJoiner j2{t2};
298
299 // Wait for the winner (with timeout).
300 auto status_future = winner_future.wait_for(std::chrono::seconds(10));
301
302 int winner_id = 0;
303 if (status_future == std::future_status::ready) {
304 winner_id = winner_future.get();
305 }
306
307 // Wait for both threads to finish their first-token read.
308 t1.join();
309 t2.join();
310
311 auto winner = (winner_id == 1) ? race1 : race2;
312 auto loser = (winner_id == 1) ? race2 : race1;
313
314 if (winner_id == 0 || !winner->success) {
315 // Neither replica produced a token.
316 if (race1->ctx) race1->ctx->TryCancel();
317 if (race2->ctx) race2->ctx->TryCancel();
318 registry_.DecrementActive(sel1->replica_id);
319 registry_.DecrementActive(sel2->replica_id);
320 queue_.NotifyCapacityAvailable();
321 queue_.NotifyCapacityAvailable();
322 return grpc::Status(grpc::StatusCode::UNAVAILABLE,
323 "both hedged replicas failed");
324 }
325
326 // Cancel the loser's stream.
327 if (loser->ctx) {
328 loser->ctx->TryCancel();
329 }
330 registry_.DecrementActive(loser->replica_id);
331 queue_.NotifyCapacityAvailable();
332 cb_manager_.RecordSuccess(winner->replica_id);
333 if (loser->success) {
334 cb_manager_.RecordSuccess(loser->replica_id); // loser also produced a token
335 }
336
337 LOG_INFO("gateway", "hedge winner: %s", winner->replica_id.c_str());
338
339 // Stream the winner's first token to the client.
340 InferResponse resp;
341 resp.set_token(winner->first_token.token());
342 resp.set_is_final(winner->first_token.is_final());
343 resp.set_replica_id(winner->replica_id);
344 writer->Write(resp);
345 int tokens_sent = 1;
346
347 // Continue streaming remaining tokens from the winner.
348 GenerateResponse gen_resp;
349 while (winner->reader->Read(&gen_resp)) {
350 if (context->IsCancelled()) {
351 winner->ctx->TryCancel();
352 break;
353 }
354
355 InferResponse infer_resp;
356 infer_resp.set_token(gen_resp.token());
357 infer_resp.set_is_final(gen_resp.is_final());
358 infer_resp.set_replica_id(winner->replica_id);
359
360 if (!writer->Write(infer_resp)) {
361 winner->ctx->TryCancel();
362 break;
363 }
364 tokens_sent++;
365 }
366
367 registry_.DecrementActive(winner->replica_id);
368 queue_.NotifyCapacityAvailable();
369
370 return grpc::Status::OK;
371}

373// ---------------------------------------------------------------------------
374// StreamFromReplica: the core streaming proxy logic.
375//
376// Flow:
377// 1. Create a gRPC channel+stub to the replica (one per call; channels
378// are lightweight and cached internally by gRPC's channel pool).
379// 2. Build a GenerateRequest with the original prompt, max_tokens, and
380// tokens_already_generated. The replica uses tokens_already_generated
381// to skip tokens that were already sent to the client by a previous
382// replica (mid-stream failover).
383// 3. Open a server-streaming RPC (Generate) and read tokens in a loop.
384// 4. For each token: check if the client cancelled, then forward the
385// token as an InferResponse on the client-facing stream. If the
386// client disconnected (Write returns false), cancel the replica
387// stream to avoid wasting GPU cycles.
388// 5. After the Read loop ends, call Finish() to get the final gRPC
389// status from the replica.
390//
391// Return value convention:
392// >= 0 : number of tokens successfully forwarded to the client.
393// -1 : replica-side error (the gRPC stream terminated with a non-OK
394// status). The caller (Infer) uses -1 to distinguish "zero
395// tokens streamed successfully" from "stream failed before any
396// token". This distinction drives the circuit-breaker recording:
397// -1 triggers RecordFailure; 0 does not.
398// ---------------------------------------------------------------------------
399int GatewayServer::StreamFromReplica(const std::string& replica_id,
400 const std::string& grpc_address,
401 const std::string& prompt,
402 int max_tokens,
403 int tokens_already_generated,
404 grpc::ServerWriter<InferResponse>* writer,
405 grpc::ServerContext* client_context) {
406 auto channel = grpc::CreateChannel(grpc_address,
407 grpc::InsecureChannelCredentials());
408 auto stub = LLMReplica::NewStub(channel);
409
410 GenerateRequest gen_request;
411 gen_request.set_request_id("req-" + std::to_string(
412 std::chrono::steady_clock::now().time_since_epoch().count()));
413 gen_request.set_prompt(prompt);
414 gen_request.set_max_tokens(max_tokens);
415 gen_request.set_tokens_already_generated(tokens_already_generated);
416
417 grpc::ClientContext replica_context;
418 auto reader = stub->Generate(&replica_context, gen_request);
419
420 int tokens_streamed = 0;
421 GenerateResponse gen_response;
422
423 while (reader->Read(&gen_response)) {
424 // Early exit if the downstream client has gone away.
425 if (client_context->IsCancelled()) {
426 replica_context.TryCancel();
427 return tokens_streamed;
428 }
429
430 // Translate the replica's GenerateResponse into the client-facing
431 // InferResponse, tagging it with the serving replica_id for
432 // observability (the client can see which replica served each token).
433 InferResponse infer_response;
434 infer_response.set_token(gen_response.token());
435 infer_response.set_is_final(gen_response.is_final());
436 infer_response.set_replica_id(replica_id);
437
438 if (!writer->Write(infer_response)) {
439 // Client disconnected mid-stream. Cancel the replica RPC to
440 // free up its resources (GPU memory, request slot).
441 replica_context.TryCancel();
442 return tokens_streamed;
443 }
444
445 tokens_streamed++;
446 }
447
448 // Finish() retrieves the final status from the replica.
449 // A non-OK status means the replica encountered an error (crash,
450 // timeout, OOM, etc.). We return -1 to signal failure so the Infer
451 // retry loop can attempt failover to another replica.
452 grpc::Status status = reader->Finish();
453 if (!status.ok()) {
454 LOG_WARN("gateway", "stream from %s failed: %s",
455 replica_id.c_str(), status.error_message().c_str());
456 return -1;
457 }
458
459 return tokens_streamed;
460}
