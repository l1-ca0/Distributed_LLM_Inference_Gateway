31 static void Write(LogLevel level, const char* component, const char* fmt, ...) {
32 if (quiet_.load()) return;
33 if (level < level_.load()) return;
34
35 // Timestamp (milliseconds since program start).
36 static auto start = std::chrono::steady_clock::now();
37 auto now = std::chrono::steady_clock::now();
38 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count();
39
40 // Level prefix.
41 const char* prefix = "";
42 switch (level) {
43 case LogLevel::DEBUG: prefix = "DBG"; break;
44 case LogLevel::INFO: prefix = "INF"; break;
45 case LogLevel::WARN: prefix = "WRN"; break;
46 case LogLevel::ERROR: prefix = "ERR"; break;
47 default: break;
48 }
49
50 // Format: [001234ms] INF [component] message
51 fprintf(stderr, "[%06ldms] %s [%s] ", static_cast<long>(ms), prefix, component);
52
53 va_list args;
54 va_start(args, fmt);
55 vfprintf(stderr, fmt, args);
56 va_end(args);

59 }
