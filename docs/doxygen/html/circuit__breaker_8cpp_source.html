10// ---------------------------------------------------------------------------
11// AllowRequest: the gating function called before dispatching to a replica.
12//
13// State-by-state behavior:
14// CLOSED: Always allow. The circuit is healthy.
15// OPEN: Block all requests until cooldown_ms has elapsed. Once the
16// cooldown expires, transition to HALF_OPEN and allow exactly
17// one probe request through. The cooldown prevents a flood of
18// retries from hammering a degraded replica.
19// HALF_OPEN: Allow one probe at a time (probe_in_flight_ guard). If a
20// probe is already in flight, block further requests until its
21// outcome is recorded. This limits exposure: we send at most
22// one request to test whether the replica has recovered.
23// ---------------------------------------------------------------------------
24bool CircuitBreaker::AllowRequest() {
25 std::lock_guard lock(mutex_);
26
27 switch (state_) {
28 case State::CLOSED:
29 return true;
30
31 case State::OPEN: {
32 // Check if cooldown has elapsed -> transition to HALF_OPEN.
33 auto now = std::chrono::steady_clock::now();
34 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
35 now - opened_at_);
36 if (elapsed.count() >= config_.cooldown_ms) {
37 state_ = State::HALF_OPEN;
38 probe_in_flight_ = true;
39 return true; // Allow one probe request.
40 }
41 return false; // Still in cooldown.
42 }
43
44 case State::HALF_OPEN:
45 // Only allow one probe at a time.
46 if (!probe_in_flight_) {
47 probe_in_flight_ = true;
48 return true;
49 }
50 return false;
51 }
52
53 return false;
54}

56// ---------------------------------------------------------------------------
57// RecordSuccess / RecordFailure: called after a request completes.
58//
59// Both methods:
60// 1. Append the outcome to the sliding window (deque).
61// 2. Trim the window to window_size (pop_front if oversized).
62// 3. Handle HALF_OPEN probe results.
63// 4. Call EvaluateState() to check for CLOSED -> OPEN transition.
64//
65// HALF_OPEN probe logic:
66// - Success: the replica has recovered. Clear the window (fresh start)
67// and close the circuit. Clearing the window avoids carrying over old
68// failures that would immediately re-trip the circuit.
69// - Failure: the replica is still degraded. Reopen the circuit with a
70// fresh cooldown timer. Skip EvaluateState() since we already know
71// the new state (OPEN).
72// ---------------------------------------------------------------------------
73void CircuitBreaker::RecordSuccess() {
74 std::lock_guard lock(mutex_);
75 outcomes_.push_back(true);
76 if (static_cast<int>(outcomes_.size()) > config_.window_size) {
77 outcomes_.pop_front();
78 }
79
80 if (state_ == State::HALF_OPEN) {
81 // Probe succeeded -> close the circuit, giving the replica a clean slate.
82 state_ = State::CLOSED;
83 probe_in_flight_ = false;
84 outcomes_.clear();
85 }
86
87 EvaluateState();
88}

90void CircuitBreaker::RecordFailure() {
91 std::lock_guard lock(mutex_);
92 outcomes_.push_back(false);
93 if (static_cast<int>(outcomes_.size()) > config_.window_size) {
94 outcomes_.pop_front();
95 }
96
97 if (state_ == State::HALF_OPEN) {
98 // Probe failed -> reopen the circuit with a fresh cooldown.
99 state_ = State::OPEN;
100 opened_at_ = std::chrono::steady_clock::now();
101 probe_in_flight_ = false;
102 return; // No need to EvaluateState; we already set OPEN.
103 }
104
105 EvaluateState();
106}

113double CircuitBreaker::error_rate() const {
114 std::lock_guard lock(mutex_);
115 if (outcomes_.empty()) return 0.0;
116 int failures = 0;
117 for (bool ok : outcomes_) {
118 if (!ok) failures++;
119 }
120 return static_cast<double>(failures) / outcomes_.size();
121}

123// ---------------------------------------------------------------------------
124// EvaluateState: check whether the sliding window error rate exceeds the
125// threshold and, if so, open the circuit.
126//
127// This only handles the CLOSED -> OPEN transition. The other transitions
128// (OPEN -> HALF_OPEN, HALF_OPEN -> CLOSED/OPEN) are handled inline in
129// AllowRequest and RecordSuccess/RecordFailure respectively.
130//
131// Guard: we require a full window (outcomes_.size() == window_size) before
132// tripping. This avoids false positives during startup when the first 1-2
133// requests might fail due to cold-start latency rather than true degradation.
134// ---------------------------------------------------------------------------
135void CircuitBreaker::EvaluateState() {
136 if (state_ != State::CLOSED) return;
137 if (static_cast<int>(outcomes_.size()) < config_.window_size) return;
138
139 int failures = 0;
140 for (bool ok : outcomes_) {
141 if (!ok) failures++;
142 }
143
144 double rate = static_cast<double>(failures) / outcomes_.size();
145 if (rate > config_.error_threshold) {
146 state_ = State::OPEN;
147 opened_at_ = std::chrono::steady_clock::now();
148 }
149}

187// GetOrCreate: lazy factory. A breaker is created on first access with the
188// shared Config. This means newly discovered replicas (via gossip) get a
189// circuit breaker automatically on their first routed request, with no
190// explicit registration step required.
191CircuitBreaker& CircuitBreakerManager::GetOrCreate(
192 const std::string& replica_id) {
193 auto it = breakers_.find(replica_id);
194 if (it == breakers_.end()) {
195 it = breakers_.emplace(replica_id, config_).first;
196 }
197 return it->second;
198}
