16// RequestQueue: FIFO backpressure queue for the inference gateway.
17//
18// When all replicas are at capacity, the gateway cannot immediately route
19// an incoming request. Instead of rejecting it with RESOURCE_EXHAUSTED,
20// the request thread blocks on this queue until a replica frees up.
21//
22// Flow:
23// 1. Gateway's Infer() handler calls SelectReplica() and gets nullopt.
24// 2. Instead of returning an error, it calls WaitForCapacity() which
25// blocks the calling thread on a condition_variable.
26// 3. When any in-flight streaming session completes, the gateway calls
27// NotifyCapacityAvailable(), which wakes one waiting thread.
28// 4. The woken thread retries SelectReplica() — it may succeed now.
29//
30// The queue is bounded (max_size_). If the queue is full when a new request
31// arrives, WaitForCapacity() returns false immediately, and the gateway
32// returns RESOURCE_EXHAUSTED. This prevents unbounded memory growth under
33// sustained overload.
34//
35// FIFO ordering: waiting threads are woken in the order they called
36// WaitForCapacity(), because condition_variable::notify_one() wakes the
37// thread that has been waiting the longest (in practice, on most
38// implementations). For strict FIFO, we use a ticket-based system.
39// ---------------------------------------------------------------------------
40class RequestQueue {
41public:
42 explicit RequestQueue(int max_size = 100);
43
44 // Block the calling thread until capacity is available or timeout expires.
45 // Returns true if the thread was woken by NotifyCapacityAvailable().
46 // Returns false if the queue is full (overload) or timeout expired.
47 bool WaitForCapacity(int timeout_ms = 30000);
48
49 // Wake one waiting thread (called when a replica finishes a request).
50 void NotifyCapacityAvailable();
51
52 // Current number of threads waiting in the queue.
53 int waiting_count() const { return waiting_count_.load(); }
54
55 // Maximum queue size.
56 int max_size() const { return max_size_; }
57
58private:
59 int max_size_;
60 std::atomic<int> waiting_count_{0};
61
62 std::mutex mutex_;
63 std::condition_variable cv_;
64
65 // Ticket system for strict FIFO ordering.
66 // Each waiting thread takes a ticket; NotifyCapacityAvailable increments
67 // the serving counter. A thread wakes only when its ticket is being served.
68 uint64_t next_ticket_ = 0;
69 uint64_t serving_ticket_ = 0;
70};
