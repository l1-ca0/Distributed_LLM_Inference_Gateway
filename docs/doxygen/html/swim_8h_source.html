22// Tuning parameters for the SWIM failure detector. Defaults are chosen so
23// detection latency (~2 * T_suspect) stays well under typical gossip test
24// budgets while keeping UDP traffic low.
25struct SwimConfig {
26 int protocol_period_ms = 500; // T_protocol: time between ping rounds
27 int ping_timeout_ms = 200; // T_ping: timeout for direct/indirect ping
28 int suspect_timeout_ms = 2000; // T_suspect: time before suspect → dead
29 int indirect_ping_count = 2; // k: number of indirect probes
30 int max_piggyback_updates = 8; // max updates piggybacked per message
31};

33// ---------------------------------------------------------------------------
34// SwimProtocol: SWIM-style gossip-based failure detector and membership
35// protocol (Das, Gupta, Motivala; "SWIM", DSN 2002).
36//
37// Each node runs two threads:
38// - Sender loop: every T_protocol, pings a random peer, and if the direct
39// ping times out, asks k=indirect_ping_count other peers to relay a
40// ping. This indirect probing separates real failures from transient
41// network drops. Also scans for SUSPECT members whose timers have
42// expired and marks them DEAD.
43// - Receiver loop: handles inbound PING, PING_REQ, and ACK messages from
44// peers, and also processes piggybacked membership updates that ride
45// on every outgoing packet.
46//
47// State propagation is fully decentralized. When a node learns of a state
48// change (locally observed or received from a peer), the update is queued
49// in the MembershipList's piggyback buffer and attached to subsequent
50// outgoing messages until it has been sent kMaxPiggybackSends times or the
51// buffer fills. No heartbeats, no central health monitor.
52//
53// Self-refutation: if a node sees a piggybacked update marking itself
54// SUSPECT/DEAD, it increments its own incarnation number and broadcasts
55// an ALIVE update at the new incarnation. Because the conflict-resolution
56// rules in MembershipList prefer the higher incarnation, the stale rumor
57// is quickly overridden.
58//
59// Thread-safety:
60// - Start()/Stop() are not thread-safe; call once.
61// - SetMembershipCallback, SetMyLoad, membership_list() accessors, and
62// is_running() are safe to call at any time from any thread.
63// - The sender and receiver threads synchronize only via the
64// MembershipList (which has its own shared_mutex) and a small ACK
65// tracking table guarded by ack_mutex_.
66// ---------------------------------------------------------------------------
