120std::vector<InferResult> InferenceClient::InferConcurrent(
121 const std::string& target_address, int count,
122 const std::string& client_id, const std::string& prompt_prefix,
123 int max_tokens, bool hedge) {
124
125 std::vector<InferResult> results(count);
126 std::vector<std::thread> threads;
127 threads.reserve(count);
128
129 for (int i = 0; i < count; i++) {
130 threads.emplace_back([&, i]() {
131 // Each thread creates its own client (own gRPC channel).
132 InferenceClient client(target_address);
133 std::string prompt = prompt_prefix + "_" + std::to_string(i);
134 results[i] = client.Infer(client_id, prompt, max_tokens, hedge);
135 results[i].request_index = i;
136 });
137 }
138
139 for (auto& t : threads) {
140 t.join();
141 }
142
143 return results;
144}
