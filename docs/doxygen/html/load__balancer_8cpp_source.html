20// SelectReplica: the hot path called on every incoming inference request.
21// It tries Tier 1 first (consistent hash for cache affinity) and only
22// falls through to Tier 2 (least-connections) if the ring is empty or

36// ---------------------------------------------------------------------------
37// RebuildRing: reconstruct the hash ring from the current alive replica set.
38//
39// Called on membership changes (replica join, leave, or state transition).
40// We snapshot the alive replicas first (shared lock on the registry),
41// then take the ring mutex to swap in the new ring. This two-step approach
42// avoids holding both locks simultaneously.
43//
44// Virtual nodes: each physical replica is placed at virtual_nodes_ positions
45// on the ring using hash(replica_id + "#" + i). This ensures that when a
46// single replica leaves, only ~1/N of the key space is remapped (consistent
47// hashing property), and the load is spread evenly rather than clumped
48// around a few hash positions.
49// ---------------------------------------------------------------------------
50void LoadBalancer::RebuildRing() {
51 auto alive = registry_.GetAliveReplicas();
52
53 std::lock_guard lock(ring_mutex_);
54 ring_.clear();
55
56 for (const auto& [id, info] : alive) {
57 for (int i = 0; i < virtual_nodes_; i++) {
58 std::string key = id + "#" + std::to_string(i);
59 size_t h = Hash(key);
60 ring_[h] = id;
61 }
62 }
63
64 LOG_DEBUG("lb", "ring rebuilt: %zu replicas, %zu points",
65 alive.size(), ring_.size());
66}

68// ---------------------------------------------------------------------------
69// Tier 1: Consistent hashing with clockwise walk.
70//
71// 1. Hash the prompt prefix (first 64 chars). Using only the prefix means
72// requests that share a common prefix (e.g., system prompt + few-shot
73// examples) will hash to the same ring position, giving stable prefix
74// affinity.
75//
76// 2. lower_bound(h) finds the first ring point >= h (clockwise from h).
77// If we overshoot the end, wrap to ring_.begin() (the ring is circular).
78//
79// 3. Walk clockwise, checking each candidate replica for:
80// - not draining (rolling update in progress)
81// - not DEAD (gossip has confirmed failure)
82// - has capacity: max_capacity == 0 means unlimited (the replica
83// self-manages concurrency), otherwise active_requests < max_capacity
84// The first replica satisfying all three is returned.
85//
86// 4. If the walk completes a full loop without finding capacity, return
87// nullopt so the caller falls through to Tier 2.
88// ---------------------------------------------------------------------------
89std::optional<LoadBalancer::Selection> LoadBalancer::SelectByConsistentHash(
90 const std::string& prompt,
91 const std::unordered_set<std::string>& excluded) {
92
93 std::lock_guard lock(ring_mutex_);
94 if (ring_.empty()) return std::nullopt;
95
96 // Hash the prompt prefix (first 64 chars or the whole prompt if shorter).
97 std::string prefix = prompt.substr(0, std::min<size_t>(64, prompt.size()));
98 size_t h = Hash(prefix);
99
100 // Find the first ring position >= h (clockwise walk).
101 auto it = ring_.lower_bound(h);
102 if (it == ring_.end()) it = ring_.begin(); // wrap around
103
104 // Walk the ring until we find a replica with capacity.
105 size_t start_hash = it->first;
106 bool wrapped = false;
107
108 while (true) {
109 const auto& replica_id = it->second;
110
111 // Skip replicas the caller has asked us to exclude (e.g., circuits
112 // open, just-failed stream). The next ring point may be a different
113 // replica; if not, we'll walk past it on the next iteration.
114 if (excluded.count(replica_id) == 0) {
115 auto info = registry_.GetReplica(replica_id);
116 // max_capacity == 0 means "unlimited": the replica has no gateway-enforced
117 // concurrency cap and self-manages its own parallelism.
118 if (info && !info->draining &&
119 info->gossip_state != llmgateway::gossip::DEAD &&
120 (info->max_capacity == 0 || info->active_requests.load() < info->max_capacity)) {
121 return Selection{replica_id, info->grpc_address};
122 }
123 }
124
125 // Move clockwise.
126 ++it;
127 if (it == ring_.end()) {
128 it = ring_.begin();
129 wrapped = true;
130 }
131
132 // Full loop -- no capacity anywhere on the ring.
133 if (wrapped && it->first >= start_hash) break;
134 }
135
136 return std::nullopt; // Ring exhausted, fall through to Tier 2.
137}

139// ---------------------------------------------------------------------------
140// Tier 2: Weighted least-connections.
141//
142// Scoring: score(r) = active_requests(r) / weight(r).
143// Dividing by weight means a replica with weight=2.0 can handle twice
144// the load of a weight=1.0 replica before its score equals theirs.
145// This naturally favors higher-capacity replicas (e.g., GPU vs CPU).
146//
147// Capacity check: replicas at max_capacity are skipped entirely.
148// max_capacity == 0 is treated as "unlimited" (same as in Tier 1).
149//
150// Tiebreak: when multiple replicas have the same minimum score, we use
151// reservoir sampling (Vitter's Algorithm R) to select uniformly at random
152// among them in a single pass. This avoids the need to collect all tied
153// candidates into a separate list. The k-th tied candidate replaces the
154// current best with probability 1/k, which yields uniform selection.
155//
156// Returns nullopt only when every alive replica is at capacity -- the
157// caller (Infer) will then return RESOURCE_EXHAUSTED to the client.
158// ---------------------------------------------------------------------------
159std::optional<LoadBalancer::Selection> LoadBalancer::SelectByLeastConnections(
160 const std::unordered_set<std::string>& excluded) {
161 auto alive = registry_.GetAliveReplicas();
162 if (alive.empty()) return std::nullopt;
163
164 std::string best_id;
165 std::string best_addr;
166 double best_score = std::numeric_limits<double>::max();
167 int tie_count = 0;
168
169 for (const auto& [id, info] : alive) {
170 if (excluded.count(id)) continue;
171 // Skip replicas that are at capacity (max_capacity > 0 means enforced).
172 if (info.max_capacity > 0 && info.active_requests.load() >= info.max_capacity) continue;
173
174 double score = static_cast<double>(info.active_requests.load()) / info.weight;
175
176 if (score < best_score) {
177 best_score = score;
178 best_id = id;
179 best_addr = info.grpc_address;
180 tie_count = 1;
181 } else if (score == best_score) {
182 // Reservoir sampling: the k-th tied candidate replaces the
183 // current winner with probability 1/k, guaranteeing uniform
184 // random selection across all tied candidates.
185 tie_count++;
186 std::uniform_int_distribution<int> dist(1, tie_count);
187 if (dist(g_rng) == 1) {
188 best_id = id;
189 best_addr = info.grpc_address;
190 }
191 }
192 }
193
194 if (best_id.empty()) return std::nullopt;
195 return Selection{best_id, best_addr};
196}
