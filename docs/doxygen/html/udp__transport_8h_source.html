15// Parsed network address used throughout the gossip layer.
16// Addresses are always "host:port" strings at rest (in protobuf fields,
17// config, and seed lists); this struct exists for comparisons and for
18// passing host/port to syscalls without re-parsing.
19struct Address {
20 std::string host;
21 uint16_t port;
22
23 std::string ToString() const { return host + ":" + std::to_string(port); }
24
25 bool operator==(const Address& other) const {
26 return host == other.host && port == other.port;
27 }
28};

37// UdpTransport: thin UDP wrapper used by the SWIM gossip layer.
38//
39// Owns a single bound UDP socket. Messages are GossipMessage protobufs
40// serialized to bytes; datagrams are assumed to fit in a single packet
41// (SWIM gossip messages with piggyback are well under the typical 64KB
42// IP datagram limit at the cluster sizes this project targets).
43//
44// Thread-safety: Send() and Receive() may be called concurrently from
45// different threads — the OS socket API is thread-safe for this pattern,
46// and the SWIM protocol uses exactly one sender thread and one receiver
47// thread. Copying and moving are disabled because the socket fd is owned.
48//
49// Lifetime: the socket is created in the constructor and closed in the
50// destructor. Construction throws std::runtime_error if bind() fails
51// (e.g., port already in use).
52// ---------------------------------------------------------------------------
53class UdpTransport {
54public:
55 explicit UdpTransport(uint16_t bind_port);
56 ~UdpTransport();
57
58 // Non-copyable, non-movable (owns a socket)
59 UdpTransport(const UdpTransport&) = delete;
60 UdpTransport& operator=(const UdpTransport&) = delete;
61
62 // Send a GossipMessage to the given address. Returns true on success.
63 bool Send(const Address& dest, const GossipMessage& message);
64
65 // Receive a GossipMessage with a timeout. Returns nullopt on timeout or error.
66 std::optional<std::pair<GossipMessage, Address>> Receive(int timeout_ms);
67
68 uint16_t port() const { return port_; }
69
70private:
71 int sock_fd_ = -1;
