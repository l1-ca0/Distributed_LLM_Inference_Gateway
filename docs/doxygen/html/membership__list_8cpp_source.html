13bool MembershipList::ApplyUpdate(const MembershipUpdate& update) {
14 std::unique_lock lock(mutex_);
15
16 const auto& id = update.member_id();
17 auto it = members_.find(id);
18
19 MemberState old_state = DEAD; // treat unknown as dead for callback purposes
20
21 if (it != members_.end()) {
22 old_state = it->second.state;
23 auto& existing = it->second;
24
25 // ---- Incarnation-based conflict resolution ----
26 // Rule 1: Higher incarnation always wins (including higher DEAD).
27 // Rule 2: At equal incarnation, stronger state wins (DEAD > SUSPECT > ALIVE).
28 // Rule 3: A strictly older incarnation is always stale and ignored,
29 // even if it claims DEAD. This is what lets a refuted node
30 // stay ALIVE: after self-refute bumps its incarnation, stale
31 // DEAD updates from piggyback buffers of uninformed peers
32 // still fail the incarnation check and don't resurrect.
33
34 if (update.incarnation() < existing.incarnation) {
35 // Stale update — ignore regardless of state. This includes stale
36 // DEAD, which would otherwise re-kill an already-refuted node.
37 return false;
38 } else if (update.incarnation() == existing.incarnation) {
39 // Same incarnation: stronger state wins.
40 // State ordering: ALIVE(0) < SUSPECT(1) < DEAD(2)
41 if (update.state() <= existing.state) {
42 // Same or weaker state — update load info only.
43 bool load_changed = false;
44 if (update.active_requests() != 0 || update.max_capacity() != 0) {
45 if (existing.active_requests != update.active_requests() ||
46 existing.max_capacity != update.max_capacity()) {
47 load_changed = true;
48 }
49 existing.active_requests = update.active_requests();
50 existing.max_capacity = update.max_capacity();
51 }
52 if (!update.model_version().empty() &&
53 existing.model_version != update.model_version()) {
54 existing.model_version = update.model_version();
55 load_changed = true;
56 }
57 // Fire the callback on a same-state load change so that
58 // downstream subscribers (e.g., the gateway's ReplicaRegistry)
59 // can refresh capacity / model-version info. Without this,
60 // subscribers would cache whatever metadata arrived with
61 // the first state-transition update forever.
62 if (load_changed && callback_) {
63 auto cb = callback_;
64 MemberState s = existing.state;
65 lock.unlock();
66 cb(id, s, s);
67 }
68 return false; // No state change.
69 }
70 // Stronger state at same incarnation — fall through to apply.
71 }
72 // Higher incarnation or stronger state — fall through to apply.
73 }
74
75 // Apply the update.
76 MemberInfo& info = members_[id];
77 MemberState new_state = update.state();
78
79 info.address = update.address();
80 info.state = new_state;
81 info.incarnation = update.incarnation();
82 if (!update.model_version().empty()) {
83 info.model_version = update.model_version();
84 }
85 if (update.active_requests() != 0 || update.max_capacity() != 0) {
86 info.active_requests = update.active_requests();
87 info.max_capacity = update.max_capacity();
88 }
89
90 if (new_state == SUSPECT) {
91 info.suspect_time = std::chrono::steady_clock::now();
92 }
93
94 // Queue for piggybacking.
95 QueueUpdate(id, info);
96
97 // Fire callback if state actually changed.
98 if (new_state != old_state && callback_) {
99 // Release lock before callback to avoid deadlocks.
100 auto cb = callback_;
101 lock.unlock();
102 cb(id, old_state, new_state);
103 }
104
105 return true;
106}

108bool MembershipList::MarkSuspect(const std::string& member_id) {
109 std::unique_lock lock(mutex_);
110 auto it = members_.find(member_id);
111 if (it == members_.end() || it->second.state != ALIVE) {
112 return false;
113 }
114
115 MemberState old_state = it->second.state;
116 it->second.state = SUSPECT;
117 it->second.suspect_time = std::chrono::steady_clock::now();
118
119 QueueUpdate(member_id, it->second);
120
121 if (callback_) {
122 auto cb = callback_;
123 lock.unlock();
124 cb(member_id, old_state, SUSPECT);
125 }
126
127 return true;
128}

198std::optional<std::pair<std::string, MemberInfo>>
199MembershipList::GetRandomAlivePeer(
200 const std::vector<std::string>& exclude) const {
201 std::shared_lock lock(mutex_);
202
203 std::vector<std::pair<std::string, MemberInfo>> candidates;
204 for (const auto& [id, info] : members_) {
205 if (id == my_id_) continue;
206 if (info.state != ALIVE && info.state != SUSPECT) continue;
207 if (std::find(exclude.begin(), exclude.end(), id) != exclude.end()) continue;
208 candidates.emplace_back(id, info);
209 }
210
211 if (candidates.empty()) return std::nullopt;
212
213 std::uniform_int_distribution<size_t> dist(0, candidates.size() - 1);
214 return candidates[dist(g_rng)];
215}

217std::vector<MembershipUpdate> MembershipList::GetUpdatesForPiggyback(
218 int max_count) {
219 std::unique_lock lock(mutex_);
220
221 std::vector<MembershipUpdate> result;
222 result.reserve(std::min(max_count, static_cast<int>(piggyback_buffer_.size())));
223
224 for (auto& entry : piggyback_buffer_) {
225 if (static_cast<int>(result.size()) >= max_count) break;
226 result.push_back(entry.update);
227 entry.send_count++;
228 }
229
230 // Remove entries that have been sent enough times.
231 while (!piggyback_buffer_.empty() &&
232 piggyback_buffer_.front().send_count >= kMaxPiggybackSends) {
233 piggyback_buffer_.pop_front();
234 }
235
236 return result;
237}

273void MembershipList::QueueUpdate(const std::string& member_id,
274 const MemberInfo& info) {
275 // Caller must hold unique_lock on mutex_.
276 MembershipUpdate update;
277 update.set_member_id(member_id);
278 update.set_address(info.address);
279 update.set_state(info.state);
280 update.set_incarnation(info.incarnation);
281 update.set_model_version(info.model_version);
282 update.set_active_requests(info.active_requests);
283 update.set_max_capacity(info.max_capacity);
284
285 // Trim buffer if too large.
286 if (piggyback_buffer_.size() >= kMaxPiggybackBuffer) {
287 piggyback_buffer_.pop_front();
288 }
289
290 piggyback_buffer_.push_back({std::move(update), 0});
291}
