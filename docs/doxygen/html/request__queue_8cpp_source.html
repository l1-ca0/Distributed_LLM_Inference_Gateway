9// ---------------------------------------------------------------------------
10// WaitForCapacity: block until notified that a replica has capacity.
11//
12// Ticket-based FIFO:
13// Each caller takes a ticket number (next_ticket_++) under the lock.
14// The condition_variable predicate checks whether serving_ticket_ has
15// reached this thread's ticket. NotifyCapacityAvailable() increments
16// serving_ticket_ and wakes all waiters, but only the one whose ticket
17// matches will pass the predicate and proceed. This guarantees strict
18// FIFO ordering regardless of OS thread scheduling.
19//
20// Bounded queue:
21// If waiting_count_ >= max_size_, the request is rejected immediately.
22// This prevents the queue from growing unboundedly under sustained
23// overload. The caller should return RESOURCE_EXHAUSTED to the client.
24//
25// Timeout:
26// A 30-second default timeout prevents threads from blocking forever
27// if the cluster is completely dead. On timeout, the thread returns
28// false and the caller should return DEADLINE_EXCEEDED or UNAVAILABLE.
29// ---------------------------------------------------------------------------
30bool RequestQueue::WaitForCapacity(int timeout_ms) {
31 std::unique_lock lock(mutex_);
32
33 // Check-and-admit happen under the lock so waiting_count_ cannot
34 // exceed max_size_ under concurrent admission.
35 if (waiting_count_.load() >= max_size_) {
36 int current = waiting_count_.load();
37 LOG_WARN("queue", "queue full (%d/%d), rejecting request",
38 current, max_size_);
39 return false;
40 }
41 waiting_count_.fetch_add(1);
42
43 uint64_t my_ticket = next_ticket_++;
44
45 auto deadline = std::chrono::steady_clock::now() +
46 std::chrono::milliseconds(timeout_ms);
47
48 // Wait until our ticket is being served, or timeout.
49 bool served = cv_.wait_until(lock, deadline, [&]() {
50 return serving_ticket_ > my_ticket;
51 });
52
53 waiting_count_.fetch_sub(1);
54
55 if (!served) {
56 LOG_WARN("queue", "request timed out after %dms in queue", timeout_ms);
57 }
58
59 return served;
60}

62// ---------------------------------------------------------------------------
63// NotifyCapacityAvailable: called when a replica finishes a request.
64//
65// Increments serving_ticket_ so the next FIFO waiter's predicate becomes
66// true, then wakes all waiters. Only the one whose ticket matches will
67// actually proceed; the rest go back to sleep.
68//
69// Using notify_all() instead of notify_one() because the predicate is
70// ticket-specific — notify_one() might wake a thread whose ticket isn't
71// next, causing it to sleep again while the correct thread remains asleep.
72// The cost is minimal (spurious wakeups are cheap; the predicate filters).
73// ---------------------------------------------------------------------------
74void RequestQueue::NotifyCapacityAvailable() {
75 {
76 std::lock_guard lock(mutex_);
77 serving_ticket_++;
78 }
79 cv_.notify_all();
80}
